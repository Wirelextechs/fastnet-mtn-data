    <script>
        // *** CRITICAL FIX: The Live Public Key is hardcoded here to bypass the failing server call (404 Not Found) ***
        let PAYSTACK_PUBLIC_KEY = 'pk_live_574b7f2d8e5798fcab3221d292ef5fa8d28c204e'; 
        let selectedPackage = null;

        // The problematic fetch('/config') block has been REMOVED.

        /**
         * Step 1: Handle package selection on the home page.
         */
        function selectPackage(element) {
            // Remove 'selected' class from all cards
            document.querySelectorAll('.package-card').forEach(card => {
                card.classList.remove('selected');
            });

            // Add 'selected' class to the clicked card
            element.classList.add('selected');
            
            // Store the data and enable the button (convert price to number)
            selectedPackage = {
                data: element.dataset.data,
                price: parseFloat(element.dataset.price)
            };
            document.getElementById('select-button').disabled = false;
        }

        /**
         * Step 2: Navigate from Home to Checkout page.
         */
        function goToCheckout() {
            if (!selectedPackage) return;

            // Update summary details on the checkout page
            document.getElementById('summary-data').textContent = selectedPackage.data + ' Non-Expiry Data';
            document.getElementById('summary-price').textContent = selectedPackage.price;

            // Toggle view visibility
            document.getElementById('home-view').style.display = 'none';
            document.getElementById('checkout-view').style.display = 'block';
            
            // Scroll to the top of the new view
            window.scrollTo(0, 0);
        }

        /**
         * Step 3: Navigate from Checkout back to Home.
         */
        function goToHome() {
            document.getElementById('checkout-view').style.display = 'none';
            document.getElementById('home-view').style.display = 'block';
            // Clear the number field for a new attempt
            document.getElementById('beneficiary-number').value = '';
            document.getElementById('customer-email').value = ''; // Clear email too
        }


        /**
         * STEP 4: The critical function for API integration.
         */
        function initiatePayment() {
            const beneficiaryNumber = document.getElementById('beneficiary-number').value.trim();
            const customerEmail = document.getElementById('customer-email').value.trim();
            
            if (!customerEmail || !customerEmail.includes('@')) {
                alert('Please enter a valid Email Address.');
                return;
            }

            // Simple check for a typical 10-digit Ghanaian number (02/05 prefix)
            if (!/^(02|05)\d{8}$/.test(beneficiaryNumber)) {
                alert('Please enter a valid 10-digit MTN Beneficiary Number (e.g., 024xxxxxxx or 054xxxxxxx).');
                return;
            }

            if (!selectedPackage) {
                alert('An error occurred. Please select a package again.');
                goToHome();
                return;
            }
            
            if (!PAYSTACK_PUBLIC_KEY) {
                 alert('Configuration Error: Paystack Public Key is missing.');
                 return;
            }

            console.log('--- Submitting Order to Backend ---');

            // 1. Send secure POST request to your backend server's /initialize endpoint
            fetch('/initialize', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({
                    // Use the captured email
                    email: customerEmail, 
                    // Send amount in the base currency (GHS)
                    amount: selectedPackage.price,
                    metadata: {
                        beneficiaryNumber: beneficiaryNumber,
                        packageData: selectedPackage.data
                    }
                })
            })
            .then(response => {
                if (!response.ok) {
                    // This handles non-200 responses from your server
                    throw new Error(`Server initialization failed with status: ${response.status}`);
                }
                return response.json();
            })
            .then(data => {
                if (data.status && data.data.reference) {

                    // 2. Use PaystackPop to handle the payment modal
                    // Calculate amount in pesewas (smallest unit for GHS)
                    const amountInPesewas = selectedPackage.price * 100;
                    
                    let handler = PaystackPop.setup({
                        key: PAYSTACK_PUBLIC_KEY, 
                        // Use the captured email
                        email: customerEmail, 
                        amount: amountInPesewas, 
                        ref: data.data.reference,
                        currency: 'GHS', 
                        metadata: {
                            custom_fields: [
                                { display_name: "Beneficiary", variable_name: "beneficiary", value: beneficiaryNumber }
                            ]
                        },
                        onClose: function(){
                            console.log('Payment window closed by user.');
                        },
                        callback: function(response){
                            const reference = response.reference;
                            // 3. Redirect to your server's /verify endpoint after success
                            window.location.href = `/verify/${reference}`; 
                        }
                    });
                    handler.openIframe();

                } else {
                    alert('Error initiating payment. Paystack initialization failed.');
                    console.error('Backend response failed:', data);
                }
            })
            .catch(error => {
                console.error('Network or Server Error:', error);
                alert('An unexpected error occurred: ' + error.message);
            });
        }
